### 关键架构决策

- **单文件全栈**: 前端 `App.tsx` 单文件组件 + 后端 `server.ts` 单文件服务，无路由拆分
- **同源部署**: Express 同时托管 API 和前端静态资源，无跨域问题
- **AI 客户端懒加载**: `getGeminiClient()` 延迟初始化，避免 API Key 缺失导致启动崩溃
- **Strict JSON Schema**: Gemini 响应强制 `responseMimeType: "application/json"` + `responseSchema`，确保输出结构可解析
- **Diff 截断保护**: 超过 60KB 的 diff 自动截断，防止 token 溢出

## 2. 后端：审计接口 (POST /api/review)

### 2.1 GitHub PR URL 解析

```typescript
function parseGithubPr(urlStr: string) {
  const url = new URL(urlStr);
  if (!url.hostname.includes("github.com")) return null;
  const parts = url.pathname.split("/").filter(Boolean);
  const pullIndex = parts.indexOf("pull");
  if (pullIndex !== -1 && parts[pullIndex + 1]) {
    return {
      owner: parts[pullIndex - 2],
      repo: parts[pullIndex - 1],
      pullNumber: parts[pullIndex + 1],
    };
  }
  return null;
}
```

### 2.2 Diff 获取（从 GitHub 拉取或手动输入）

```typescript
if (url && !targetDiff) {
  const parsed = parseGithubPr(url);
  if (!parsed) return res.status(400).json({ error: "Invalid GitHub PR URL" });

  const diffUrls = [
    `https://patch-diff.githubusercontent.com/raw/${owner}/${repo}/pull/${pullNumber}.diff`,

  for (const dUrl of diffUrls) {
    const fetchRes = await fetch(dUrl, {
      headers: { "Accept": "text/plain", "User-Agent": "AI-Studio-PR-Reviewer" },

    if (fetchRes.ok) { targetDiff = await fetchRes.text(); break; }

  if (!targetDiff) return res.status(502).json({ error: "Could not fetch diff" });

if (!targetDiff || !targetDiff.trim()) {

if (isDiffTruncated) {
```

## 3. 前端：Diff 解析与注释匹配

### 3.1 Unified Diff 解析

```typescript
const parsedFiles = useMemo(() => {
  const diffText = reviewResult?.diff || "";
  const filesList: { fileName: string; lines: { type: 'add'|'delete'|'normal'|'header'; content: string; oldLine?: number; newLine?: number }[] }[] = [];
  let currentFile = null, oldLineNum = 0, newLineNum = 0;

  for (const line of diffText.split("\n")) {
    if (line.startsWith("diff --git ")) {
      const match = line.match(/b\/(.+)$/);
      currentFile = { fileName: match ? match[1] : "", lines: [] };
      filesList.push(currentFile);
      oldLineNum = 0; newLineNum = 0;
    } else if (currentFile) {
      if (line.startsWith("@@ ")) {
        const hunkMatch = line.match(/^@@ -(\d+),?\d* \+(\d+),?\d* @@/);
        if (hunkMatch) { oldLineNum = parseInt(hunkMatch[1]); newLineNum = parseInt(hunkMatch[2]); }
      } else if (line.startsWith("+")) {
        currentFile.lines.push({ type: 'add', content: line.substring(1), newLine: newLineNum++ });
      } else if (line.startsWith("-")) {
        currentFile.lines.push({ type: 'delete', content: line.substring(1), oldLine: oldLineNum++ });
      } else {
        currentFile.lines.push({ type: 'normal', content: line, oldLine: oldLineNum++, newLine: newLineNum++ });
      }
    }
  }
  return filesList;
}, [reviewResult]);
```

## 4. 前端：审计触发与状态管理

```typescript
const handleStartReview = async (e: React.FormEvent) => {
  e.preventDefault();
  if (!prUrl && !pastedDiff) {
    setErrorMessage("请输入 GitHub PR URL 或粘贴 Diff 文本");
    return;
  }

  setLoading(true);
  setErrorMessage("");

  try {
    const response = await fetch("/api/review", {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({ url: prUrl, diff: pastedDiff || null }),
    });
    const data = await response.json();
    if (!response.ok) throw new Error(data.error || "服务器处理异常");

    setReviewResult(data);
    setResolvedIssues({});
  } catch (err: any) {
    setErrorMessage(err.message || "无法调用大模型审计");
  } finally {
    setLoading(false);
  }
};
```
