### 关键架构决策

- **单文件全栈**: 前端 `App.tsx` 单文件组件 + 后端 `server.ts` 单文件服务，无路由拆分
- **同源部署**: Express 同时托管 API 和前端静态资源，无跨域问题
- **两阶段分析管线**: Gemini 提示词重构为 `Phase 1: 业务语义 → Phase 2: 代码审计` 的强制顺序执行
- **"软约束 + 硬兜底"**: 语义字段不加入 schema `required`，通过自然语言强制声明 + 代码层 `inferSemanticsFromDiff()` 兜底
- **三层语义保障**: Gemini 输出 → hasEmptySemantics 检测 → 本地启发式推理，确保语义字段永远有值
- **AI 客户端懒加载**: `getGeminiClient()` 延迟初始化，避免 API Key 缺失导致启动崩溃
- **Diff 截断保护**: 超过 50KB 的 diff 自动截断，防止 token 溢出
- **环境变量 fallback**: `.env` → `.env.example` → 系统环境变量 的三级加载策略
- **零依赖代理**: `undici` ProxyAgent 全局拦截 `fetch`，支持 `HTTPS_PROXY` 环境变量

### 2.2 GitHub PR URL 解析

```typescript
function parseGithubPr(urlStr: string) {
  const url = new URL(urlStr);
  if (!url.hostname.includes("github.com")) return null;

  const pathname = url.pathname.replace(/\/+$/, "");
  const parts = pathname.split("/").filter(Boolean);

  const pullIndex = parts.indexOf("pull");
  if (pullIndex !== -1 && parts[pullIndex + 1] && /^\d+$/.test(parts[pullIndex + 1])) {
    const pullNumber = parts[pullIndex + 1];
    if (pullIndex >= 2) {
      return { owner: parts[pullIndex - 2], repo: parts[pullIndex - 1], pullNumber };
    }
  }

  // 正则回退解析 (处理嵌套仓库、查询参数等非标准格式)
  const match = urlStr.match(/github\.com\/([^\/]+)\/([^\/]+)\/pull\/(\d+)/);
  if (match) {
    return { owner: match[1], repo: match[2], pullNumber: match[3] };
  }
  return null;
}
```

### 2.3 Diff 获取与清洗

```typescript
// 带超时的 fetch
async function fetchWithTimeout(url: string, options: RequestInit = {}, timeout = 10000) {
  const controller = new AbortController();
  const timeoutId = setTimeout(() => controller.abort(), timeout);
  try {
    return await fetch(url, { ...options, signal: controller.signal });
  } finally {
    clearTimeout(timeoutId);
  }
}

// 八进制转义序列解码 (解决中文文件名乱码)
function decodeGitOctalEscapes(diffText: string): string {
  return diffText.replace(/"([^"]*\\[0-7]{3}[^"]*)"/g, (match) => {
    const inner = match.slice(1, -1);
    const bytes: number[] = [];
    let i = 0;
    while (i < inner.length) {
      if (inner[i] === '\\' && /^[0-7]{3}$/.test(inner.slice(i + 1, i + 4))) {
        bytes.push(parseInt(inner.slice(i + 1, i + 4), 8));
        i += 4;
      } else {
        bytes.push(inner.charCodeAt(i));
        i++;
      }
    }
    return '"' + new TextDecoder('utf-8').decode(new Uint8Array(bytes)) + '"';
  });
}

function cleanDiffContent(diffText: string): string {
  let cleaned = decodeGitOctalEscapes(diffText);
  cleaned = cleaned.replace(/\r\n/g, "\n").replace(/\n{3,}/g, "\n\n").trim();
  return cleaned;
}

// Unified Diff 格式验证
function isValidUnifiedDiff(diffText: string): boolean {
  const lines = diffText.trim().split('\n');
  if (lines.length === 0 || !lines[0].startsWith('diff --git ')) return false;

  let hasFileHeader = false, hasHunkHeader = false, hasContent = false;
  for (const line of lines) {
    if (line.startsWith('--- ') || line.startsWith('+++ ')) hasFileHeader = true;
    else if (line.startsWith('@@ -') && /^@@ -(\d+),?\d* \+(\d+),?\d* @@/.test(line))
      hasHunkHeader = true;
    else if ((line.startsWith('+') || line.startsWith('-')) && hasHunkHeader)
      hasContent = true;
  }
  return hasFileHeader && hasHunkHeader && hasContent;
}

// Diff 获取主流程 (支持粘贴 Diff 优先)
if (url && !hasPastedDiff) {
  const diffUrls = [
    `https://patch-diff.githubusercontent.com/raw/${owner}/${repo}/pull/${pullNumber}.diff`,
    `https://github.com/${owner}/${repo}/pull/${pullNumber}.diff`,
  ];

  let errorType: 'network' | 'auth' | 'not_found' | 'other' = 'other';
  for (const dUrl of diffUrls) {
    try {
      const fetchRes = await fetchWithTimeout(dUrl, { /* ... */ }, 15000);
      if (fetchRes.ok) { targetDiff = await fetchRes.text(); break; }
      if (fetchRes.status === 401 || fetchRes.status === 403) errorType = 'auth';
      else if (fetchRes.status === 404) errorType = 'not_found';
    } catch (e: any) {
      if (e.name === 'AbortError') errorType = 'network';
    }
  }
}

// Diff 获取失败 → 返回高保真 Mock + 本地语义推理
if (!targetDiff) {
  fallbackResult.diff = mockDiff;
  fallbackResult.codeBlocks = detectCodeBlocks(fallbackResult.diff);
  fallbackResult.semantics = inferSemanticsFromDiff(fallbackResult.diff, fallbackResult.codeBlocks);
  return res.json({ ...fallbackResult, fetchError: { type, message, details } });
}

// 格式验证
if (!isValidUnifiedDiff(targetDiff)) {
  return res.status(400).json({ error: "Invalid Git Diff format" });
}

// Diff 截断保护
targetDiff = cleanDiffContent(targetDiff);
```

### 2.4 PR 元数据获取

```typescript
async function fetchPrMetadata(owner: string, repo: string, pullNumber: string) {
  const token = process.env.GITHUB_TOKEN;
  const headers: Record<string, string> = {
    "Accept": "application/vnd.github.v3+json",
    "User-Agent": "AI-Studio-PR-Reviewer",
  };
  if (token) headers["Authorization"] = `Bearer ${token}`;

  const prUrl = `https://api.github.com/repos/${owner}/${repo}/pulls/${pullNumber}`;
  const prRes = await fetchWithTimeout(prUrl, { headers }, 10000);
  if (!prRes.ok) throw new Error(`GitHub API status ${prRes.status}`);

  const prData = await prRes.json();
  const issueRefs = [...((prData.body || '') as string).matchAll(/#(\d+)/g)];

  return {
    owner, repo, pullNumber: parseInt(pullNumber, 10),
    title: prData.title || '', description: (prData.body || '').substring(0, 2000),
    author: prData.user?.login || '', createdAt: prData.created_at || '',
    linkedIssues: issueRefs.map(m => ({ number: parseInt(m[1]), title: '' })),
  };
}
```

```typescript
  const patterns = [
```

### 2.8 响应后处理与语义兜底

### 3.1 Unified Diff 解析 (含八进制解码)

```typescript
function decodeGitOctalEscapes(diffText: string): string { /* 同后端逻辑 */ }

const parsedFiles = useMemo(() => {
  const diffText = decodeGitOctalEscapes(reviewResult?.diff || "");
  const filesList: { fileName: string; lines: DiffLine[] }[] = [];
  let currentFile = null, oldLineNum = 0, newLineNum = 0;

  for (const line of diffText.split("\n")) {
    if (line.startsWith("diff --git ")) {
      const match = line.match(/b\/(.+)$/);
      currentFile = { fileName: match ? match[1] : "", lines: [] };
      filesList.push(currentFile);
      oldLineNum = 0; newLineNum = 0;
      currentFile.lines.push({ type: 'header', content: line });
    } else if (currentFile) {
      if (line.startsWith("--- ") || line.startsWith("+++ ") || line.startsWith("index ")) {
        currentFile.lines.push({ type: 'header', content: line });
      } else if (line.startsWith("@@ ")) {
        currentFile.lines.push({ type: 'header', content: line });
        const hunkMatch = line.match(/^@@ -(\d+),?\d* \+(\d+),?\d* @@/);
        if (hunkMatch) { oldLineNum = parseInt(hunkMatch[1]); newLineNum = parseInt(hunkMatch[2]); }
      } else if (line.startsWith("+")) {
        currentFile.lines.push({ type: 'add', content: line.substring(1), newLine: newLineNum++ });
      } else if (line.startsWith("-")) {
        currentFile.lines.push({ type: 'delete', content: line.substring(1), oldLine: oldLineNum++ });
      } else {
        currentFile.lines.push({ type: 'normal', content: line, oldLine: oldLineNum++, newLine: newLineNum++ });
      }
    }
  }
  return filesList;
}, [reviewResult]);
```
